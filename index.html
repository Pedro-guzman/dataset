# Grfica nmero de sservicios por fecha
# Nos aseguramos que la colunma Fecha tenga el formato correcto
df['Fecha'] = pd.to_datetime(df['Fecha'], errors='coerce')

# Agrupamos los servicios por fecha y los contamos
servicio_por_fecha = df.groupby('Fecha').size()

# Grfica
servicio_por_fecha.plot(kind='line', marker='o')
plt.title('Nmero de servicios por fecha')
plt.xlabel('Fecha')
plt.ylabel('Cantidad de servicios')
plt.xticks(servicio_por_fecha.index, servicio_por_fecha.index.strftime('%Y-%m-%d'), rotation=90)
plt.grid(True)
plt.show()
